- $ docker ps -a

- $ docker stop sys-container-2 
- $ docker rm sys-container-1

- $ docker run -it --name=sys-container-1 centos:7 bash
- $ docker run -it --name=sys-container-2 ubuntu:16.04 bash

hub.docker.com에 로그인 후 이미지 확인하고 필요한 이미지 다운 받아서 사용할 것

nignx images를 통해 구현 가능한 컨데이터(웹서버) 
nignx 대표기술 - Web Server, Reverse proxy, API Gateway(alpine 버전이 좀 더 경량임)
- $ docker pull nginx:1.25.0-alpine

열린포트 확인 방법
- $ docker image history nginx:1.25.0-alpine

구동 후 해당 url 기준으로 접속 가능 192.168.56.101:8001
- $ docker run -d -p 8001:80 --name=my-webserver1 nginx:1.25.0-alpine

port 확인
- $ docker port my-webserver1
- $ sudo netstat -nlp | grep 8001

파일복사
- $ docker cp index.html my-webserver1:/usr/share/nginx/html/index.html

사용했던 소스를 그대로 만드려면
- $ vi Dockerfile
- 도커 파일 내용
FROM nginx:1.25.0-alpine
COPY index-image.html /usr/share/nginx/html/index.html
COPY docker_logo.png /usr/share/nginx/html/docker_logo.png
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

도커 이미지 생성
- $ docker build -t myweb:v1.0 .

잘 동작하는지 확인
- $ docker run -d -p 8002:80 --name=my-webserver2 myweb:v1.0
